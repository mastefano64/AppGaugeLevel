// Funzione per calcolare lo stile in base alla temperatura
function styleFunction(feature) {
  const temperature = feature.get('temperatura');

  // Calcoliamo un valore normalizzato tra 0 e 1
  const normalizedTemperature = (temperature - minTemperature) / (maxTemperature - minTemperature);

  // Creiamo un colore interpolato tra blu e rosso
  const color = [0, 0, 255]; // Blu
  color[0] = Math.round(255 * normalizedTemperature); // Aumenta il rosso gradualmente
  color[2] = Math.round(255 * (1 - normalizedTemperature)); // Diminuisce il blu gradualmente

  return new ol.style.Style({
    fill: new ol.style.Fill({
      color: 'rgb(' + color.join(',') + ')'
    })
  });
}
